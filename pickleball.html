function generateRound() { const MAX_SITOUT = 2; const slots = courts * 4; /* ----- Initialize state ----- */ players.forEach(p => { playCount[p] ??= 0; sitoutStreak[p] ??= 0; partnerCount[p] ??= {}; }); /* ===================================================== PHASE 1 — FORCE PLAYERS WHO SAT OUT TOO LONG ===================================================== */ const forcedPlayers = players .filter(p => sitoutStreak[p] >= MAX_SITOUT) .sort((a, b) => playCount[a] - playCount[b]); let playing = [...forcedPlayers]; /* If more forced players than slots, trim by fairness */ if (playing.length > slots) { playing = playing.slice(0, slots); } /* ===================================================== PHASE 2 — FILL REMAINING SLOTS FAIRLY ===================================================== */ const remainingSlots = slots - playing.length; if (remainingSlots > 0) { const remainingCandidates = players .filter(p => !playing.includes(p)) .sort((a, b) => { if (playCount[a] !== playCount[b]) { return playCount[a] - playCount[b]; } return sitoutStreak[b] - sitoutStreak[a]; }); /* Controlled randomness buffer */ const buffer = Math.min( 2, Math.max(0, remainingCandidates.length - remainingSlots) ); const candidatePool = remainingCandidates .slice(0, remainingSlots + buffer) .sort(() => Math.random() - 0.5); playing.push(...candidatePool.slice(0, remainingSlots)); } /* ===================================================== PHASE 3 — UPDATE SIT‑OUT STREAKS ===================================================== */ players.forEach(p => { if (playing.includes(p)) { sitoutStreak[p] = 0; } else { sitoutStreak[p]++; } }); /* ===================================================== PHASE 4 — PARTNER‑AWARE PAIRING ===================================================== */ let matches = []; let pool = [...playing]; while (pool.length >= 4) { let bestCombo = null; let bestScore = Infinity; for (let i = 0; i < pool.length; i++) { for (let j = i + 1; j < pool.length; j++) { for (let k = 0; k < pool.length; k++) { for (let l = k + 1; l < pool.length; l++) { const set = new Set([i, j, k, l]); if (set.size < 4) continue; const a = pool[i], b = pool[j]; const c = pool[k], d = pool[l]; const score = (partnerCount[a][b] || 0) + (partnerCount[c][d] || 0); if (score < bestScore) { bestScore = score; bestCombo = [a, b, c, d]; } } } } } const [a, b, c, d] = bestCombo; matches.push({ teamA: [a, b], teamB: [c, d] }); /* Commit partner history */ partnerCount[a][b] = (partnerCount[a][b] || 0) + 1; partnerCount[b][a] = partnerCount[a][b]; partnerCount[c][d] = (partnerCount[c][d] || 0) + 1; partnerCount[d][c] = partnerCount[c][d]; /* Commit play counts */ playCount[a]++; playCount[b]++; playCount[c]++; playCount[d]++; /* Remove used players */ pool = pool.filter(p => ![a, b, c, d].includes(p)); } return { matches }; }
